SwiftyMessenger API

Freely connect your iOS, macOS, or other Swift project to Facebook Messenger using Macenger's SwiftyMessenger API client framework.

Login to Messenger

import SwiftyMessenger

let client = SwiftyMessenger()

let myUsername = "zarkmuckerberg@gmail.com"
let myPassword = "********"

client.login(username: myUsername, password: myPassword) { res in
	if res != .success {
		throw myApp.loginError
	}
}

Send message in thread

import SwiftyMessenger

let client = SwiftyMessenger(...)

let myMessage = SentMessageData(text: "Hello, World!")
let thread = 729040210304

client.send(message: myMessage, threadId: thread)

Fetch thread messages

import SwiftyMessenger

let client = SwiftyMessenger(...)

let thread = 729040210304

client.fetchThreadMessages(thread) { res in
	do {
		let messages = try res.unwrap().messagesData!.messages
		for msg in messages {
			print(msg.message)
		}
	} catch {
		// ...
	}
}
